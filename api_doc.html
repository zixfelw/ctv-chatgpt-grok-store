import requests

API_URL = "http://103.69.87.202:5000"
API_KEY = "DLR_xxxxx"   # ← Thay bằng key thật

def get_stock():
    """Lấy danh sách sản phẩm + stock + giá."""
    try:
        r = requests.get(
            f"{API_URL}/api/dealer/stock",
            headers={"X-API-KEY": API_KEY},
            timeout=10  # ← QUAN TRỌNG: luôn đặt timeout
        )
        data = r.json()

        if data.get("success"):
            balance = data["balance"]
            products = data["products"]
            print(f"Số dư: {balance:,}đ")
            for key, info in products.items():
                print(f"  {key}: {info['name']} | Giá: {info['price']:,}đ | Stock: {info['stock']}")
            return products
        else:
            print(f"Lỗi: {data.get('error')}")
            return None

    except requests.exceptions.ConnectionError:
        print("❌ Không kết nối được server! Kiểm tra lại URL hoặc server đang down")
        return None
    except requests.exceptions.Timeout:
        print("❌ Server không phản hồi (timeout). Thử lại sau")
        return None
    except Exception as e:
        print(f"❌ Lỗi không xác định: {e}")
        return None

def buy_product(product_key, qty):
    """Mua sản phẩm thường (GPTPlus, CapCut, v.v.)."""
    try:
        r = requests.post(
            f"{API_URL}/api/dealer/buy",
            headers={
                "X-API-KEY": API_KEY,
                "Content-Type": "application/json"   # ← BẮT BUỘC
            },
            json={
                "product_key": product_key,
                "qty": qty
            },
            timeout=30
        )
        data = r.json()

        if data.get("success"):
            items = data["items"]          # ← Danh sách tài khoản
            charged = data["total_charged"]
            remaining = data["balance_remaining"]
            order_code = data["order_code"]
            print(f"✅ Mua thành công {qty}x {product_key}")
            print(f"   Trừ: {charged:,}đ | Còn: {remaining:,}đ")
            print(f"   Mã đơn: {order_code}")
            for item in items:
                print(f"   → {item}")
            return items
        else:
            error = data.get("error", "Lỗi không xác định")
            print(f"❌ Mua thất bại: {error}")
            return None

    except requests.exceptions.ConnectionError:
        print("❌ Không kết nối được server!")
        return None
    except requests.exceptions.Timeout:
        print("❌ Timeout — server không phản hồi")
        return None
    except Exception as e:
        print(f"❌ Lỗi: {e}")
        return None

# Sử dụng:

if items:
    # Gửi items cho khách hàng của bạn

def buy_slot_gpt_team(emails):

        return None

def safe_buy(product_key, qty, emails=None):
    """Mua hàng với xử lý lỗi đầy đủ."""
    try:
        body = {"product_key": product_key, "qty": qty}
        if emails:
            body["emails"] = emails

        r = requests.post(
            f"{API_URL}/api/dealer/buy",
            headers={
                "X-API-KEY": API_KEY,
                "Content-Type": "application/json"
            },
            json=body,
            timeout=180 if product_key == "slot_gpt_team" else 30
        )

        # Kiểm tra HTTP status trước
        if r.status_code == 401:
            return None, "Thiếu API key! Kiểm tra header X-API-KEY"
        elif r.status_code == 403:
            return None, "API key sai! Gõ /ctv lấy key mới"
        elif r.status_code == 404:
            return None, f"Sản phẩm '{product_key}' không tồn tại"

        data = r.json()

        if data.get("success"):
            return data["items"], None   # ← Thành công
        else:
            return None, data.get("error", "Lỗi không rõ")

    except requests.exceptions.ConnectionError:
        return None, "Không kết nối được server (server down hoặc sai URL)"
    except requests.exceptions.Timeout:
        return None, "Server không phản hồi (timeout)"
    except ValueError:
        return None, "Server trả về dữ liệu không hợp lệ (không phải JSON)"
    except Exception as e:
        return None, f"Lỗi: {str(e)}"

# Sử dụng:
items, error = safe_buy("gptplus_1thang_KBH", 1)
if error:
    print(f"❌ {error}")
    # → Thông báo lỗi cho khách
else:
    print(f"✅ Mua thành công!")
    for item in items:
        print(f"  → {item}")
    # → Gửi items cho khách

Ví dụ tích hợp vào bot Telegram Python (python-telegram-bot).

import requests
from telegram import Update, InlineKeyboardButton, InlineKeyboardMarkup
from telegram.ext import ApplicationBuilder, CommandHandler, CallbackQueryHandler, ContextTypes

# ============================
# CẤU HÌNH — THAY ĐỔI Ở ĐÂY
# ============================
BOT_TOKEN = "YOUR_BOT_TOKEN"
API_URL = "http://103.69.87.202:5000"
API_KEY = "DLR_xxxxx"  # ← Key từ /ctv

HEADERS = {
    "X-API-KEY": API_KEY,
    "Content-Type": "application/json"
}

# ============================
# HÀM GỌI API
# ============================
def api_get_stock():
    """Lấy danh sách sản phẩm từ API đối tác."""
    try:
        r = requests.get(
            f"{API_URL}/api/dealer/stock",
            headers={"X-API-KEY": API_KEY},
            timeout=10
        )
        data = r.json()
        if data.get("success"):
            return data["products"], data["balance"]
        return None, 0
    except Exception as e:
        print(f"ERROR - Lỗi lấy stock API đối tác: {e}")
        return None, 0

def api_buy(product_key, qty, emails=None):
    """Mua hàng từ API đối tác."""
    try:
        body = {"product_key": product_key, "qty": qty}
        if emails:
            body["emails"] = emails

        r = requests.post(
            f"{API_URL}/api/dealer/buy",
            headers=HEADERS,
            json=body,
            timeout=180 if product_key == "slot_gpt_team" else 30
        )
        return r.json()
    except requests.exceptions.ConnectionError:
        return {"success": False, "error": "Không kết nối được server đối tác"}
    except requests.exceptions.Timeout:
        return {"success": False, "error": "Timeout - server đối tác không phản hồi"}
    except Exception as e:
        return {"success": False, "error": str(e)}

# ============================
# HANDLER BOT TELEGRAM
# ============================
async def menu(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Hiển thị menu sản phẩm."""
    products, balance = api_get_stock()
    if products is None:
        await update.message.reply_text("❌ Server đối tác đang bận, thử lại sau!")
        return

    buttons = []
    for key, info in products.items():
        stock = info["stock"]
        if stock == 0:
            status = "❌ Hết hàng"
        elif stock == -1:
            status = "🔄 Đang cập nhật"
        else:
            status = f"✅ Còn {stock}"

        buttons.append([InlineKeyboardButton(
            f"{info['name']} | {info['price']:,}đ | {status}",
            callback_data=f"buy_{key}"
        )])

    await update.message.reply_text(
        f"🛒 *Menu sản phẩm*\n💰 Số dư: *{balance:,}đ*",
        parse_mode="Markdown",
        reply_markup=InlineKeyboardMarkup(buttons)
    )

async def handle_buy(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Xử lý khi khách nhấn mua."""
    query = update.callback_query
    await query.answer()

    product_key = query.data.replace("buy_", "")

    # Mua 1 sản phẩm
    result = api_buy(product_key, 1)

    if result.get("success"):
        items = result["items"]
        charged = result["total_charged"]
        items_text = "\n".join([f"• `{item}`" for item in items])
        await query.edit_message_text(
            f"✅ *Mua thành công!*\n\n"
            f"📦 Sản phẩm:\n{items_text}\n\n"
            f"💰 Đã trừ: *{charged:,}đ*",
            parse_mode="Markdown"
        )
    else:
        await query.edit_message_text(
            f"❌ *Lỗi:* {result.get('error', 'Không rõ')}"
        )

# ============================
# CHẠY BOT
# ============================
app = ApplicationBuilder().token(BOT_TOKEN).build()
app.add_handler(CommandHandler("menu", menu))
app.add_handler(CallbackQueryHandler(handle_buy, pattern="^buy_"))
app.run_polling()

❓ Câu hỏi thường gặp / Xử lý sự cố

🔴 "Không kết nối được server" / ConnectionError

Nguyên nhân: Server đang down hoặc firewall chặn port 5000.
Cách kiểm tra: Mở trình duyệt → vào http://103.69.87.202:5000/api/dealer/stock (sẽ trả lỗi 401 nhưng nghĩa là server đang chạy).
Xử lý: Liên hệ admin (Telegram: @longdt8386) để kiểm tra server.

🔴 "Body JSON không hợp lệ" khi POST /buy

Nguyên nhân: Thiếu header Content-Type: application/json.
Xử lý: Luôn gửi kèm header: 

headers = {
    "X-API-KEY": "DLR_xxxxx",
    "Content-Type": "application/json"   # ← QUAN TRỌNG!
}

🔴 stock = -1 cho slot_gpt_team

Nguyên nhân: Hệ thống đang cập nhật slot (tự động mỗi 60 giây).
Xử lý: Chờ 60 giây rồi gọi lại /api/dealer/stock. KHÔNG hiển thị "hết hàng" cho khách khi stock = -1!
